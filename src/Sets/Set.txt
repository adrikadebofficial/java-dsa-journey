// HashSet notes

import java.util.HashSet;
import java.util.Iterator;

//declare
        HashSet<Integer> set = new HashSet<>();

//Insert
        set.add(1);
        set.add(2);
        set.add(3);

//Search -- contains
       if(set.contains(1)){
           System.out.println("Set contains element ");
       }else{
           System.out.println("Doesnot contain element");
       }

//Delete -- remove

        set.remove(1);
       if(!set.contains(1)){
           System.out.println("Doesnot contain 1");
       }

//Length --size
        System.out.println("size of set is: " + set.size());

//Print all elements
        System.out.println(set);

//Iterator
        Iterator it = set.iterator();

//hasNext & next function
        while(it.hasNext()){
            System.out.println(it.next());
        }
